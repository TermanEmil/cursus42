NAME		=	ft_ls
FLAGS		=	-Wall -Wextra -Werror
CC			=	gcc $(FLAGS)

SRC_PROJ	=	$(shell find src -name '*.c')
SRC			=	$(SRC_PROJ)

INC_LIBFT	=	$(shell find libft/includes -type d)
INC_LIBFT_P	=	$(foreach d, $(INC_LIBFT), -I$d)

INC_DIRS	=	$(shell find includes -type d)
INC_PROJ	=	$(foreach d, $(INC_DIRS), -I$d)

INCLUDES	=	$(INC_PROJ) $(INC_LIBFT_P)

OBJ			=	*.o
OBJDIR		=	./obj/
OBJS		=	$(addprefix $(OBJDIR), $(OBJ))

ifeq ($(OS),Windows_NT)
	RM = del
	RMRF = rd /s /q
	MV = move
	MKDIR = mkdir
else
	RM = rm -f
	RMRF = rm -rf
	MV = mv
	MKDIR = mkdir -p
endif

all: $(NAME)

$(NAME):
		@gcc $(SRC) $(INCLUDES) libft/libft.a -o $(NAME)	
#@gcc .main.c $(INCLUDES) $(SRC) libft/libft.a -o rs

clean:
		

fclean: clean
		@$(RM) $(NAME)

re: fclean all

main:
	@gcc $(INCLUDES) $(NAME) -o rs

run:
	@./$(NAME)

sure: re run
